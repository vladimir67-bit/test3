per_cent = {'ТКБ': 5.6, 'СКБ': 5.9, 'ВТБ': 4.28, 'СБЕР': 4.0}
S = float(input("Введите сумму"))
D = []                             #создаем пустой список
for i in list(per_cent.values()):  #берем поочередно значения из словаря
    D.append((i * S)/100)          #вычисляем прибыль каждого значения и дополняем в список D
print("deposit =", D)              #выводим список
print("Максимальная сумма, которую вы можете заработать -", max(D)) #выводим наибольшее значение
